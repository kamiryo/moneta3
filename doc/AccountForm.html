def amount_less
  return unless payment
  if payment.amount + payment.minus_limit < amount
    errors.add :amount, "残高が不足しています"
  end
end

def save
  return false if invalid?
  if deposit.save && payment
    payment.withdrow(amount) && deposit.deposit(amount)
  else
    false
  end
end
